# flake.nix — crate2nix + rust-overlay (via flake-parts)

Generate a `flake.nix` and supporting Nix modules for per-crate granular Nix builds.

## flake.nix

```nix
{
  description = "{{description}}";

  nixConfig = {
    extra-substituters = [ "https://crate2nix.cachix.org" ];
    extra-trusted-public-keys = [ "crate2nix.cachix.org-1:bXMeMOBI39htMnFaFj5MkBczuNKDfTwBBzHbPmcJ+lE=" ];
  };

  inputs = {
    nixpkgs.url = "github:NixOS/nixpkgs/nixpkgs-unstable";
    flake-parts.url = "github:hercules-ci/flake-parts";
    rust-overlay = {
      url = "github:oxalica/rust-overlay";
      inputs.nixpkgs.follows = "nixpkgs";
    };
    crate2nix.url = "github:nix-community/crate2nix";
    devshell = {
      url = "github:numtide/devshell";
      inputs.nixpkgs.follows = "nixpkgs";
    };
  };

  outputs = inputs:
    inputs.flake-parts.lib.mkFlake { inherit inputs; } {
      systems = [ "x86_64-linux" "aarch64-linux" "x86_64-darwin" "aarch64-darwin" ];

      imports = [
        inputs.devshell.flakeModule
        ./nix/rust-overlay/flake-module.nix
        ./nix/devshell/flake-module.nix
      ];

      perSystem = { pkgs, system, ... }:
        let
          cargoNix = inputs.crate2nix.tools.${system}.appliedCargoNix {
            name = "{{project-name}}";
            src = ./.;
          };
        in
        {
          checks = {
            {{#if workspace}}
            # For workspaces, build all workspace members
            {{#each crates}}
            {{this.check-name}} = cargoNix.workspaceMembers.{{this.name}}.build;
            {{/each}}
            {{else}}
            # For single crates, build the root crate
            {{project-check-name}} = cargoNix.rootCrate.build;
            {{/if}}
          };

          packages = {
            {{#if has-binary}}
            default = cargoNix.workspaceMembers.{{binary-crate}}.build;
            {{/if}}
          };
        };
    };
}
```

## nix/rust-overlay/flake-module.nix

```nix
{ inputs, ... }:
{
  perSystem = { system, ... }: {
    _module.args.pkgs = import inputs.nixpkgs {
      inherit system;
      overlays = [ inputs.rust-overlay.overlays.default ];
    };
  };
}
```

## nix/devshell/flake-module.nix

```nix
{ inputs, ... }:
{
  perSystem = { pkgs, ... }:
    let
      rust-toolchain = pkgs.rust-bin.fromRustupToolchainFile ../../rust-toolchain.toml;
    in
    {
      devshells.default = {
        packages = [
          rust-toolchain
          pkgs.just
          pkgs.cargo-watch
          {{#if insta}}
          pkgs.cargo-insta
          {{/if}}
        ];

        env = [
          { name = "RUST_BACKTRACE"; value = "1"; }
        ];
      };
    };
}
```

## rust-toolchain.toml

```toml
[toolchain]
channel = "stable"
components = ["rustfmt", "clippy"]
```

## .envrc

```
use flake
```

## Directory structure for Nix files

```
project/
├── flake.nix
├── flake.lock              # generated by `nix flake lock`
├── rust-toolchain.toml
├── .envrc
└── nix/
    ├── rust-overlay/
    │   └── flake-module.nix
    └── devshell/
        └── flake-module.nix
```

## Notes

- **rust-overlay** provides the Rust toolchain from `rust-toolchain.toml` via `rust-bin.fromRustupToolchainFile`
- **crate2nix** generates per-crate Nix derivations from `Cargo.toml`/`Cargo.lock`
- **flake-parts** splits the flake into composable modules under `nix/`
- **devshell** (numtide) provides the development shell with Rust toolchain and tools
- For workspaces: use `cargoNix.workspaceMembers.<name>.build` (not `rootCrate`)
- For single crates: use `cargoNix.rootCrate.build`
- The `nixConfig` section adds the crate2nix cachix for faster builds
- After generating, run `nix flake lock` to create the lock file
- The `.envrc` with `use flake` integrates with direnv for automatic shell activation
